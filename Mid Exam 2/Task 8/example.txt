cmake_minimum_required(VERSION 3.14)
project(ExprCalculator CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(calc
    main.cpp
    Token.cpp
    Lexer.cpp
    Parser.cpp
    ASTNode.cpp
    SymbolTable.cpp
    VM.cpp
    ICalculator.cpp
    StatementNode.cpp
    StatementParser.cpp
    Manager.cpp
    Traverser.cpp
)

target_include_directories(calc PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
